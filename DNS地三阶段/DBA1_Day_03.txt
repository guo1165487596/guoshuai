数据导入导出：
	搜索路径：（默认检索目录）
		查看搜索路径：show variables like "%file%";
		修改搜索路径：vim /etc/my.cnf
			       secure_file_priv=/myfile		##修改默认路径
			       :wq
			       mkdir /myfile
			       chown mysql /myfile		##修改权限属组
			       systemctl restart mysqld
			       show variables like "%file%";	##再次查看路径

	数据导入：（把系统文件的内容存储到数据库的表里）
		命令格式：load data infile "目录/文件"" into table 库名.表名
			fields terminated by "分隔符"
			lines terminated by "\n";
			
			load data infile "/myfile/passwd" into table wg.t3
			fields terminated by ":"
			lines terminated by "\n";
	
	数据导出：（把表记录存储到系统文件里）
		命令格式1：select命令 into outfile "目录/文件";
			select * from wg.t3 into outfile "/myfile/1.txt";
		命令格式2：select命令 into outfile "目录/文件" fields terminated by "##";
			select * from wg.t3 into outfile "/myfile/2.txt" fields terminated by "##";
		命令格式3：select命令 into outfile "目录/文件" fields terminated by "##" lines terminated by "??":
			select * from wg.t3 into outfile "/myfile/3.txt" fields terminated by "##" lines terminated by "??";


管理表记录：
	添加记录：
		命令格式1：添加1条记录，给所有字段赋值，字段名可以省略不写
			insert into wg.t3 values(......);
		命令格式2：添加N条记录，给所有字段赋值
			insert into wg.t3 values(......),(.......),(.......);
		命令格式3：添加1条记录，给指定字段赋值，需要写字段名
			insert into wg.t3(......) values(......);	##前后字段需要对应
		命令格式4：添加N条记录，给指定字段赋值，需要写字段名
			insert into wg.t3(......) values(......),(......),(......);
	查看记录：
		命令格式1：查看所有记录
			select 字段1......字段N from 库名.表名；
			select * from wg.t3;
		命令格式2：条件查询
			select 字段1......字段N from 库名.表名 where 条件表达式；
			select * from wg.t3 where id=1;
	修改记录：
		命令格式1：批量更新
			update 库名.表名 set 字段名=值，字段名=值，字段名=值；
			update wg.t3 set name=wg,user=wg;
		命令格式2：条件匹配更新
			update 库名.表名 set 字段名=值，字段名=值，where 匹配条件；
			update wg.t3 set name=wg,user=wg where id=1;
	删除记录：
		命令格式1：删除所有记录
			delete from 库名.表名；
			delete from wg.t3;
		命令格式2：条件匹配删除
			delete from 库名.表名 where 条件表达式；
			delete from wg.t3 where id=1;

匹配条件：
	基本匹配条件：适用于select \ update \ delete
		数值比较：字段必须是数值类型
			=	##等于
			>	##大于
			>=	##大于或等于
			<	##小于
			<=	##小于或等于
			!=	##不相等
		字符比较：字段必须是字符类型
			=		##相等
			!=		##不相等
			is null		##空
			is not null		##非空
		范围内比较：匹配范围内的任意一个值即可
			in (值列表)		##在...里...
				select * from wg.t3 where shell in ("/bin/bash","/sbin/nologin");
			not in (值列表)		##不在...里...
				select * from wg.t3 where shell not in ("/bin/bash","/sbin/nologin");
			between 数字 and 数字	##在...之间...
				select * from wg.t3 where id between 1 and 10;
			distince 字段名		##去重显示
				select distinct shell from wg.t3;
		逻辑比较：多个条件判断时使用
			or		##逻辑或（多个条件判断，某个成立即可）
			and		##逻辑与（多个条件判断，必须同时都成立）
			!或not		##逻辑非（取反或not）
			( )		##提高优先级

	高级匹配条件：适用于select \ update \ delete 
		模糊匹配：where 字段名 like '通配符'
			_ 		##表示一个字符
			%		##表示0~n个字符
			例子：select * from wg.t3 where name like "r_o_";

		正则匹配：where 字段名 regexp '正则表达式'
			^ $ . [ ] * |		##正则元字符
			例子：select * from  wg.t3 where name regrexp '^j|y$';
		
		四则计算：字段必须是数值类型
			+	##加法
			-	##减法
			*	##乘法
			/	##除法
			%	##取余数（求模）
			( )	##提高优先级	
			select name,uid,gid,uid+gid 别名 from wg.t3;		##求总和
			select name,uid,gid,(uid+gid)/2 别名 from wg.t3;	##求平均数
	
	操作查询结果：
		聚合函数：mysql内置数据统计函数
			avg(字段名)	##统计字段平均值
			sum(字段名)	##统计字段之和
			min(字段名)	##统计字段最小值
			max(字段名)	##统计字段最大值
			count(字段名)	##统计字段值个数
			select count(*) from wg.t3;
		查询结果排序：sql查询 order by 字段名 [asc|desc];
			asc	##升序排序
			desc	##降序排序
			select name,uid from wg.t3 where uid>=10 and uid<=500 order by uid desc;		##降序排序
		查询结果分组：sql查询 group by 字段名；
			select shell from wg.t3 where uid<=500 group by shell;
			select uid from wg.t3 group by uid;
		查询结果过滤：sql查询  having 条件表达式；
			select name from wg.t3 where id<=20 having name="adm";
		限制查询结果显示行数：
			格式1：sql查询 limit 数字；		##显示查询结果前多少条记录
				select name,uid,gid from wg.t3 limit 3;
			格式2：sql查询 limit 数字1，数字2；	##显示指定范围内的查询记录
				select name,uid,gid from wg.t3 limit 3,3;
				（数字1：表示起始行，数字2表示总行数）

Mysql管理方式：
	思路：1、部署环境LAMP
	          2、启动httpd服务程序
	          3、解压phpMyAdmin包，并移动到网站目录
	          4、配置config.inc.php，指定MySQL服务器地址
	          5、浏览器访问、使用数据库管理员root登陆
		yum -y install httpd php php-mysql httpd
		systemctl restart httpd
		systemctl enable httpd		##搭建LAMP环境
		tar -xf phpMyAdmin-2.11.11-all-languages.tar.gz
		mv  phpMyAdmin-2.11.11-all-languages /var/www/html/phpmyadmin
		cd /var/www/html/phpmyadmin/
		cp config.sample.inc.php config.inc.php		##创建主配置文件
		vim config.inc.php
